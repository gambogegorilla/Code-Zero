class Solution {
    public double findMaxAverage(int[] nums, int k) {
        double curr_val = 0, max_val = 0;

        for (int i = 0; i < k; i++) {
            curr_val += nums[i];
        }

        max_val = curr_val;

        for (int i = 0; i < nums.length - k; i++) {
            curr_val -= nums[i];
            curr_val += nums[i + k];
            max_val = Math.max(curr_val, max_val);
        }

        double res = (double) max_val / k;

        return res;
    }
}
